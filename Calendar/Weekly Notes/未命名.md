function isInThisWeek(days){
	return page => {
	  let creationTime = Date.parse(page.file.ctime);
	  let currentTime = new Date().getTime();
	  let daysAgo = currentTime - days*24*60*60*1000;
	  return creationTime >= daysAgo;
	}
}
